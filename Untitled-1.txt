# Create a PowerPoint presentation for the hackathon solution
from pptx import Presentation

prs = Presentation()

slides_content = [
    ("Title Slide", "Hackathon Solution: Turning a Loss-Making Company into a Profitable Business\nRole of CMR (Customer/Company Relationship Manager)"),
    ("Problem Statement", "The company is facing continuous losses due to declining sales, poor customer satisfaction, and increasing operational costs."),
    ("Key Challenges", "1. Decreasing customer retention\n2. Lack of customer feedback analysis\n3. Inefficient operations\n4. Poor marketing strategy"),
    ("Role of CMR", "A CMR manages relationships with customers, analyzes feedback, and ensures customer needs are met to increase loyalty and revenue."),
    ("Proposed Solution", "Develop a technology-driven system that analyzes customer data, identifies problems, and suggests strategies to improve sales and satisfaction."),
    ("Solution Components", "1. Customer Feedback Analysis Tool\n2. Sales & Performance Dashboard\n3. Customer Retention Strategies\n4. Cost Optimization Insights"),
    ("Technology Used", "Data Analytics, AI-based Sentiment Analysis, Cloud Dashboard, CRM Integration"),
    ("Expected Outcomes", "1. Improved customer satisfaction\n2. Increased sales\n3. Reduced operational costs\n4. Long-term profitability"),
    ("Conclusion", "By leveraging customer insights and technology, the CMR can identify issues early and implement strategies that transform losses into profits.")
]

for title, content in slides_content:
    slide_layout = prs.slide_layouts[1] if title != "Title Slide" else prs.slide_layouts[0]
    slide = prs.slides.add_slide(slide_layout)
    slide.shapes.title.text = title
    if slide_layout != prs.slide_layouts[0]:
        slide.placeholders[1].text = content
    else:
        slide.placeholders[1].text = content

file_path = "/mnt/data/hackathon_cmr_solution.pptx"
prs.save(file_path)
